document.addEventListener('DOMContentLoaded', () => { const songSelect = document.getElementById('songSelect'); const audioFileInput = document.getElementById('audioFile'); const audioPlayer = document.getElementById('audioPlayer'); // Predefined song paths (you'll need to replace with actual paths) const songs = { 'song1': 'path/to/all_my_love.mp3', 'song2': 'path/to/iaam.mp3', 'song3': 'path/to/the_scientist.mp3' }; // Update source when song is selected songSelect.addEventListener('change', () => { audioPlayer.src = songs[songSelect.value]; audioPlayer.load(); }); // Update source when file is uploaded audioFileInput.addEventListener('change', (event) => { const file = event.target.files[0]; audioPlayer.src = URL.createObjectURL(file); audioPlayer.load(); }); // Set initial song audioPlayer.src = songs['song1']; });
